        with db.connection:
            cursor = db.connection.cursor(db.pymysql.cursors.DictCursor)
            cursor.execute("""
            SELECT
                id,
                username,
                public_key
            FROM jwt_users
            WHERE username = %s""", username)

            user = cursor.fetchone()
            return JwtUser(**user) if user else None
